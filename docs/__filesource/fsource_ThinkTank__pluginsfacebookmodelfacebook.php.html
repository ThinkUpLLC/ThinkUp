  51.   /*
  52.    * Create a Facebook client like this:
  53.    *
  54.    * $fb = new Facebook(API_KEY, SECRET);
  55.    *
  56.    * This will automatically pull in any parameters, validate them against the
  57.    * session signature, and chuck them in the public $fb_params member variable.
  58.    *
  59.    * @param api_key                  your Developer API key
  60.    * @param secret                   your Developer API secret
  61.    * @param generate_session_secret  whether to automatically generate a session
  62.    *                                 if the user doesn't have one, but
  63.    *                                 there is an auth token present in the url,
  64.    */
  65.   public function __construct($api_key, $secret, $generate_session_secret=false) {
  66.     $this->api_key                 = $api_key;
  67.     $this->secret                  = $secret;
  68.     $this->generate_session_secret = $generate_session_secret;
  69.     $this->api_client = new FacebookRestClient($api_key, $secret, null);
  70.     $this->validate_fb_params();
  71.  
  72.     // Set the default user id for methods that allow the caller to
  73.     // pass an explicit uid instead of using a session key.
  74.     $defaultUser = null;
  75.     if ($this->user) {
  76.       $defaultUser = $this->user;
  77.     } else if ($this->profile_user) {
  78.       $defaultUser = $this->profile_user;
  79.     } else if ($this->canvas_user) {
  80.       $defaultUser = $this->canvas_user;
  81.     }
  82.  
  83.     $this->api_client->set_user($defaultUser);
  84.  
  85.  
  86.     if (isset($this->fb_params['friends'])) {
  87.       $this->api_client->friends_list =
  88.         array_filter(explode(',', $this->fb_params['friends']));
  89.     }
  90.     if (isset($this->fb_params['added'])) {
  91.       $this->api_client->added = $this->fb_params['added'];
  92.     }
  93.     if (isset($this->fb_params['canvas_user'])) {
  94.       $this->api_client->canvas_user = $this->fb_params['canvas_user'];
  95.     }
  96.   }

  98.   /*
  99.    * Validates that the parameters passed in were sent from Facebook. It does so
  100.    * by validating that the signature matches one that could only be generated
  101.    * by using your application's secret key.
  102.    *
  103.    * Facebook-provided parameters will come from $_POST, $_GET, or $_COOKIE,
  104.    * in that order. $_POST and $_GET are always more up-to-date than cookies,
  105.    * so we prefer those if they are available.
  106.    *
  107.    * For nitty-gritty details of when each of these is used, check out
  108.    * http://wiki.developers.facebook.com/index.php/Verifying_The_Signature
  109.    *
  110.    * @param bool  resolve_auth_token  convert an auth token into a session
  111.    */
  112.   public function validate_fb_params($resolve_auth_token=true) {
  113.     $this->fb_params = $this->get_valid_fb_params($_POST, 48 * 3600, 'fb_sig');
  114.  
  115.     // note that with preload FQL, it's possible to receive POST params in
  116.     // addition to GET, so use a different prefix to differentiate them
  117.     if (!$this->fb_params) {
  118.       $fb_params = $this->get_valid_fb_params($_GET, 48 * 3600, 'fb_sig');
  119.       $fb_post_params = $this->get_valid_fb_params($_POST, 48 * 3600, 'fb_post_sig');
  120.       $this->fb_params = array_merge($fb_params, $fb_post_params);
  121.     }
  122.  
  123.     // Okay, something came in via POST or GET
  124.     if ($this->fb_params) {
  125.       $user               = isset($this->fb_params['user']) ?
  126.                             $this->fb_params['user'] : null;
  127.       $this->profile_user = isset($this->fb_params['profile_user']) ?
  128.                             $this->fb_params['profile_user'] : null;
  129.       $this->canvas_user  = isset($this->fb_params['canvas_user']) ?
  130.                             $this->fb_params['canvas_user'] : null;
  131.       $this->base_domain  = isset($this->fb_params['base_domain']) ?
  132.                             $this->fb_params['base_domain'] : null;
  133.       $this->ext_perms    = isset($this->fb_params['ext_perms']) ?
  134.                             explode(',', $this->fb_params['ext_perms'])
  135.                             : array();
  136.  
  137.       if (isset($this->fb_params['session_key'])) {
  138.         $session_key =  $this->fb_params['session_key'];
  139.       } else if (isset($this->fb_params['profile_session_key'])) {
  140.         $session_key =  $this->fb_params['profile_session_key'];
  141.       } else {
  142.         $session_key = null;
  143.       }
  144.       $expires     = isset($this->fb_params['expires']) ?
  145.                      $this->fb_params['expires'] : null;
  146.       $this->set_user($user,
  147.                       $session_key,
  148.                       $expires);
  149.     }
  150.     // if no Facebook parameters were found in the GET or POST variables,
  151.     // then fall back to cookies, which may have cached user information
  152.     // Cookies are also used to receive session data via the Javascript API
  153.     else if ($cookies =
  154.              $this->get_valid_fb_params($_COOKIE, null, $this->api_key)) {
  155.  
  156.       $base_domain_cookie = 'base_domain_' . $this->api_key;
  157.       if (isset($_COOKIE[$base_domain_cookie])) {
  158.         $this->base_domain = $_COOKIE[$base_domain_cookie];
  159.       }
  160.  
  161.       // use $api_key . '_' as a prefix for the cookies in case there are
  162.       // multiple facebook clients on the same domain.
  163.       $expires = isset($cookies['expires']) ? $cookies['expires'] : null;
  164.       $this->set_user($cookies['user'],
  165.                       $cookies['session_key'],
  166.                       $expires);
  167.     }
  168.     // finally, if we received no parameters, but the 'auth_token' GET var
  169.     // is present, then we are in the middle of auth handshake,
  170.     // so go ahead and create the session
  171.     else if ($resolve_auth_token && isset($_GET['auth_token']) &&
  172.              $session = $this->do_get_session($_GET['auth_token'])) {
  173.       if ($this->generate_session_secret &&
  174.           !empty($session['secret'])) {
  175.         $session_secret = $session['secret'];
  176.       }
  177.  
  178.       if (isset($session['base_domain'])) {
  179.         $this->base_domain = $session['base_domain'];
  180.       }
  181.  
  182.       $this->set_user($session['uid'],
  183.                       $session['session_key'],
  184.                       $session['expires'],
  185.                       isset($session_secret) ? $session_secret : null);
  186.     }
  187.  
  188.     return !empty($this->fb_params);
  189.   }

  221.   // Invalidate the session currently being used, and clear any state associated

  236.   /** Logs the user out of all temporary application sessions as well as their

  252.   /**
  253.    *  Clears any persistent state stored about the user, including
  254.    *  cookies and information related to the current session in the
  255.    *  client.
  256.    *
  257.    */
  258.   public function clear_cookie_state() {
  259.     if (!$this->in_fb_canvas() && isset($_COOKIE[$this->api_key . '_user'])) {
  260.        $cookies = array('user', 'session_key', 'expires', 'ss');
  261.        foreach ($cookies as $name) {
  262.          setcookie($this->api_key . '_' . $name,
  263.                    false,
  264.                    time() - 3600,
  265.                    '',
  266.                    $this->base_domain);
  267.          unset($_COOKIE[$this->api_key . '_' . $name]);
  268.        }
  269.        setcookie($this->api_key, false, time() - 3600, '', $this->base_domain);
  270.        unset($_COOKIE[$this->api_key]);
  271.      }
  272.  
  273.      // now, clear the rest of the stored state
  274.      $this->user = 0;
  275.      $this->api_client->session_key = 0;
  276.   }

  338.                            $required_permissions));

  406.   public function set_cookies($user, $session_key, $expires=null, $session_secret=null) {

  445.   /*
  446.    * Get the signed parameters that were sent from Facebook. Validates the set
  447.    * of parameters against the included signature.
  448.    *
  449.    * Since Facebook sends data to your callback URL via unsecured means, the
  450.    * signature is the only way to make sure that the data actually came from
  451.    * Facebook. So if an app receives a request at the callback URL, it should
  452.    * always verify the signature that comes with against your own secret key.
  453.    * Otherwise, it's possible for someone to spoof a request by
  454.    * pretending to be someone else, i.e.:
  455.    *      www.your-callback-url.com/?fb_user=10101
  456.    *
  457.    * This is done automatically by verify_fb_params.
  458.    *
  459.    * @param  assoc  $params     a full array of external parameters.
  460.    *                            presumed $_GET, $_POST, or $_COOKIE
  461.    * @param  int    $timeout    number of seconds that the args are good for.
  462.    *                            Specifically good for forcing cookies to expire.
  463.    * @param  string $namespace  prefix string for the set of parameters we want
  464.    *                            to verify. i.e., fb_sig or fb_post_sig
  465.    *
  466.    * @return  assoc the subset of parameters containing the given prefix,
  467.    *                and also matching the signature associated with them.
  468.    *          OR    an empty array if the params do not validate
  469.    */
  470.   public function get_valid_fb_params($params, $timeout=null, $namespace='fb_sig') {
  471.     $prefix = $namespace . '_';
  472.     $prefix_len = strlen($prefix);
  473.     $fb_params = array();
  474.     if (empty($params)) {
  475.       return array();
  476.     }
  477.  
  478.     foreach ($params as $name => $val) {
  479.       // pull out only those parameters that match the prefix
  480.       // note that the signature itself ($params[$namespace]) is not in the list
  481.       if (strpos($name, $prefix) === 0) {
  482.         $fb_params[substr($name, $prefix_len)] = self::no_magic_quotes($val);
  483.       }
  484.     }
  485.  
  486.     // validate that the request hasn't expired. this is most likely
  487.     // for params that come from $_COOKIE
  488.     if ($timeout && (!isset($fb_params['time']) || time() - $fb_params['time'] > $timeout)) {
  489.       return array();
  490.     }
  491.  
  492.     // validate that the params match the signature
  493.     $signature = isset($params[$namespace]) ? $params[$namespace] : null;
  494.     if (!$signature || (!$this->verify_signature($fb_params, $signature))) {
  495.       return array();
  496.     }
  497.     return $fb_params;
  498.   }

  513.   /**
  514.    * Validates that a given set of parameters match their signature.
  515.    * Parameters all match a given input prefix, such as "fb_sig".
  516.    *
  517.    * @param $fb_params     an array of all Facebook-sent parameters,

  525.   /**
  526.    * Validate the given signed public session data structure with
  527.    * public key of the app that
  528.    * the session proof belongs to.
  529.    *
  530.    * @param $signed_data the session info that is passed by another app
  531.    * @param string $public_key Optional public key of the app. If this
  532.    *                is not passed, function will make an API call to get it.
  533.    *  return true if the session proof passed verification.
  534.    */
  535.   public function verify_signed_public_session_data($signed_data,
  536.                                                     $public_key = null) {
  537.  
  538.     // If public key is not already provided, we need to get it through API
  539.     if (!$public_key) {
  540.       $public_key = $this->api_client->auth_getAppPublicKey(
  541.         $signed_data['api_key']);
  542.     }
  543.  
  544.     // Create data to verify
  545.     $data_to_serialize = $signed_data;
  546.     unset($data_to_serialize['sig']);
  547.     $serialized_data = implode('_', $data_to_serialize);
  548.  
  549.     // Decode signature
  550.     $signature = base64_decode($signed_data['sig']);
  551.     $result = openssl_verify($serialized_data, $signature, $public_key,
  552.                              OPENSSL_ALGO_SHA1);
  553.     return $result == 1;
  554.   }

  556.   /*
  557.    * Generate a signature using the application secret key.
  558.    *
  559.    * The only two entities that know your secret key are you and Facebook,
  560.    * according to the Terms of Service. Since nobody else can generate
  561.    * the signature, you can rely on it to verify that the information
  562.    * came from Facebook.
  563.    *
  564.    * @param $params_array   an array of all Facebook-sent parameters,
  565.    *                        NOT INCLUDING the signature itself
  566.    * @param $secret         your app's secret key
  567.    *
  568.    * @return a hash to be checked against the signature provided by Facebook
  569.    */
  570.   public static function generate_sig($params_array, $secret) {
  571.     $str = '';
  572.  
  573.     ksort($params_array);
  574.     // Note: make sure that the signature parameter is not already included in
  575.     //       $params_array.
  576.     foreach ($params_array as $k=>$v) {
  577.       $str .= "$k=$v";
  578.     }
  579.     $str .= $secret;
  580.  
  581.     return md5($str);
  582.   }
